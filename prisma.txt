// This is your Prisma schema file,
// learn more about it in the docs: https://pris.ly/d/prisma-schema

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model User {
  id           Int        @id @default(autoincrement())
  email        String     @unique
  passwordHash String
  createdAt    DateTime   @default(now())
  responses    Response[]

  @@map("users")
}

model Survey {
  id          Int        @id @default(autoincrement())
  title       String
  description String?
  createdAt   DateTime   @default(now())
  questions   Question[]
  responses   Response[]

  @@map("surveys")
}

model Question {
  id          Int      @id @default(autoincrement())
  surveyId    Int
  title       String
  description String?
  type        String
  required    Boolean  @default(false)
  orderIndex  Int
  options     String?  // JSON string of options for select/checkbox questions
  survey      Survey   @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  answers     Answer[]

  @@map("questions")
}

model Response {
  id          Int      @id @default(autoincrement())
  userId      Int
  surveyId    Int
  submittedAt DateTime @default(now())
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  survey      Survey   @relation(fields: [surveyId], references: [id], onDelete: Cascade)
  answers     Answer[]

  @@map("responses")
}

model Answer {
  id         Int      @id @default(autoincrement())
  responseId Int
  questionId Int
  value      String
  response   Response @relation(fields: [responseId], references: [id], onDelete: Cascade)
  question   Question @relation(fields: [questionId], references: [id], onDelete: Cascade)

  @@map("answers")
}
